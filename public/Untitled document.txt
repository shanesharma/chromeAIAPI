export function editElementCSS(el, str)
{
    el.style.cssText =  str;
}


export function focusOnElement(el)
{
    const tagType = el.tagName()

    if(tagType == "TEXTAREA" || tagType ==  "INPUT"){el.focus() }
    return true;

}



export function makeInputAlert(str)
{
